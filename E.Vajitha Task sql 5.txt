use productsdb;

create table customer_orders (id int primary key, item_name varchar(100), quantity int, price decimal(10,2));

insert into customer_orders ( id, item_name, quantity, price)
		values (1, 'Mobiles', 2, 20000.00),
               (2, 'Earbuds', 5, 1500.00),
			   (3, 'Smartwatch', 3, 5000.00),
               (4, 'Charger', 4, 800.00),
               (5, 'Power Bank', 2, 1200.00);
               
select * from customer_orders;

-- 1)

select count(*) as total_orders from customer_orders;

-- 2)

select sum(price) as total_price from customer_orders;

-- 3)

select avg(price) as average_price from customer_orders;

-- 4) 

select item_name, sum(price) as total_sales from customer_orders group by item_name;

-- 5)

select max(price) as higher_price, min(price) as lowest_price from customer_orders;

-- 6)

select item_name, sum(price) as total_sales from customer_orders group by item_name order by total_sales desc;
