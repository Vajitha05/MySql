use company_db;

-- 1)

create table sales (id int primary key, product varchar(50), amount decimal(10,2));
insert into sales (id, product, amount)
		values(1, 'Laptop', 50000),
			  (2, 'Mobile', 30000),
			  (3, 'Tablet', 30000),
			  (4, 'Monitor', 20000),
			  (5, 'Keyboard', 10000);	

select * from sales;

-- 2)

select id, product, amount, row_number() over (order by amount desc) as row_num from sales;

-- 3)

select id, product, amount, rank() over (order by amount desc) as rank_num from sales;

-- 4)

select id, product, amount, dense_rank() over (order by amount desc) as dense_rank_num from sales;

-- 5)

select id, product, amount, sum(amount) over (order by  id) as running_total from sales;

-- 6)

with AvgSales as (select avg(amount) as avg_amount from sales) 
			select s.id, s.product, s.amount from sales s join AvgSales a on s.amount > a.avg_amount;
